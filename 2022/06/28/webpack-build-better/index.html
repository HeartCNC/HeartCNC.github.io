webpack打包优化，js分包，css分包

2022年6月28日
  • webpack

需求：降低打包后最大文件的体积

webpack v3​

分离JS​

module.exports = {
  ...,
  plugins: [
    new webpack.optimize.CommonsChunkPlugin({
      name: 'chunk-elementUI',
      minChunks: (module) => module.resource &&
      module.resource.startWidth(path.join(__dirname, '../node_modules/element-ui'))
      }
    })
  ]
}

分离CSS​

引入外部插件 css-split-webpack-plugin

const CSSSplitWebpackPlugin = require('css-split-webpack-plugin').default

module.exports = {
  ...,
  plugins: [
    new ExtractTextPlugin({
      filename: utils.assetsPath('css/[name].[contenthash].css'),
      allChunks: true,
    }),
    new CSSSplitWebpackPlugin({ size: 4000 })
  ]
}

webpack v4/v5​

分离JS​

config.optimization.splitChunks({
  chunks: 'all',
  cacheGroups: {
    libs: {
      name: 'chunk-libs',
      test: /[\\/]node_modules[\\/]/,
      priority: 10,
      chunks: 'initial'
    },
    elementUI: {
      name: 'chunk-elementUI',
      priority: 20,
      test: /[\\/]node_modules[\\/]_?element-ui(.*)/
    },
    commons: {
      name: 'chunk-commons',
      test: resolve('src/components'),
      minChunks: 3,
      priority: 5,
      reuseExistingChunk: true
    }
  }
})
